\begin{frame}[fragile]
\frametitle{Documento}
\framesubtitle{Como inserir uma figura no documento}
  \scriptsize
  \begin{columns}[c]
  \column{.5\textwidth}
  \begin{verbatim}
\begin{figure}[h!]
  \centering
  \label{fig:tux}
  \includegraphics[width=0.5\textwidth]
                     {334px-tuxsvg.png}
  \caption{Tux.}
\end{figure}
  \end{verbatim}
  \column{.5\textwidth}
  \begin{fmpage}{\textwidth}
\begin{figure}[h!]
  \centering
  \label{fig:tux}
    \includegraphics[width=0.5\textwidth]{figures/334px-tuxsvg.png}
  \caption{Tux.}
\end{figure}
  \end{fmpage}
  \end{columns}
\end{frame}
